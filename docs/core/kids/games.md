# Simple Environmental Game Framework
class EcoGame:
    def __init__(self):
        self.resources = 100
        self.sustainability = 0
        
    def plant_tree(self):
        self.resources -= 10
        self.sustainability += 5
        
    def recycle(self):
        self.resources += 15
        self.sustainability += 3
